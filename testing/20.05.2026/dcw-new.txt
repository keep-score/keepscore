def http_get_json(url, timeout=4):
    req = urllib.request.Request(url, headers={"User-Agent": "Mozilla/5.0"})
    with urllib.request.urlopen(req, timeout=timeout) as response:
        raw = response.read().decode("utf-8", errors="replace")
        return json.loads(raw)

def http_get_text(url, timeout=4):
    req = urllib.request.Request(url, headers={"User-Agent": "Mozilla/5.0"})
    with urllib.request.urlopen(req, timeout=timeout) as response:
        return response.read().decode("utf-8", errors="replace")

def http_post_json(url, data_dict, headers=None, timeout=6):
    encoded = urllib.parse.urlencode(data_dict).encode("utf-8")

    final_headers = {
        "User-Agent": "Mozilla/5.0",
        "Accept": "application/json, text/javascript, */*; q=0.01",
        "Content-Type": "application/x-www-form-urlencoded; charset=UTF-8",
        "X-Requested-With": "XMLHttpRequest",
    }

    if headers:
        final_headers.update(headers)

    req = urllib.request.Request(url, data=encoded, headers=final_headers, method="POST")
    with urllib.request.urlopen(req, timeout=timeout) as response:
        raw = response.read().decode("utf-8", errors="replace")
        return json.loads(raw)

def build_camera_query(params):

def clean_single_line(text):
    text = str(text or "")
    text = text.replace("\r\n", " ").replace("\r", " ").replace("\n", " ").replace("|", " ")
    text = re.sub(r"\s+", " ", text).strip()
    return text

def clean_for_dahua(text):
    text = str(text or "")
    text = text.replace("\r\n", " ")
    text = text.replace("\r", " ")
    text = text.replace("\n", " ")
    text = text.replace("|", " ")
    text = text.replace("'", " ")
    text = text.replace("’", " ")
    return text

def get_first_nonempty(content, keys, default=""):
    for key in keys:
        val = content.get(key)
        if val is not None and str(val) != "":
            return val
    return default

def parse_scores_from_display(display):
    text = str(display or "")
    m = re.search(r"(\d+)\s*[-:]\s*(\d+)", text)
    if m:
        return m.group(1), m.group(2)

    nums = re.findall(r"\d+", text)
    if len(nums) >= 2:
        return nums[0], nums[1]

    return "0", "0"

def build_player_line(score, player_name, breaker_for_this_line):
    try:
        score_num = int(score)
    except Exception:
        score_num = 0

    score_text = str(score_num)
    player_name = clean_for_dahua(player_name)

    if score_num <= 9:
        score_text = "  " + score_text

    line = f"{score_text}  {player_name}".rstrip()

    if breaker_for_this_line:
        line = f"{line} •".rstrip()

    return line if line.strip() else "."

def weighted_name_length_for_box(text):
    text = clean_for_dahua(text).upper()

    total = 0.0
    for ch in text:
        if ch == ".":
            total += 0.5
        elif ch == "I":
            total += 0.5
        elif ch == " ":
            total += 0.6
        else:
            total += 1.0

    return total

def compute_name_box_right(player1_name, player2_name):
    p1_weight = weighted_name_length_for_box(player1_name)
    p2_weight = weighted_name_length_for_box(player2_name)

    longest_weight = max(p1_weight, p2_weight)
    effective_len = longest_weight + 1.0

    right = NAME_BOX_LEFT + int(effective_len * NAME_BOX_PER_CHAR) + NAME_BOX_EXTRA

    if right < NAME_BOX_MIN_RIGHT:
        right = NAME_BOX_MIN_RIGHT
    if right > NAME_BOX_MAX_RIGHT:
        right = NAME_BOX_MAX_RIGHT

    return right

def get_name_box_rects(player1_name, player2_name):
    right = compute_name_box_right(player1_name, player2_name)
    p1_rect = (NAME_BOX_LEFT, 110, right, 450)
    p2_rect = (NAME_BOX_LEFT, 560, right, 900)
    return p1_rect, p2_rect

def normalize_match_text(text):
    text = str(text or "")
    text = unicodedata.normalize("NFD", text)
    text = "".join(ch for ch in text if unicodedata.category(ch) != "Mn")
    return text.lower().strip()

def parse_match_target(match_info):
    text = normalize_match_text(match_info)

    patterns = [
        r"\bpremier\s+a\s+(\d{1,3})\b",
        r"\bpremiere?\s+a\s+(\d{1,3})\b",
        r"\bfirst\s+to\s+(\d{1,3})\b",
        r"\brace\s+to\s+(\d{1,3})\b",
        r"\bcourse\s+a\s+(\d{1,3})\b",
    ]

    for pattern in patterns:
        m = re.search(pattern, text, flags=re.IGNORECASE)
        if m:
            try:
                value = int(m.group(1))
                if value > 0:
                    return value
            except Exception:
                pass

    return None

def get_winner_info(score1, score2, match_info, player1, player2):
    try:
        s1 = int(score1)
    except Exception:
        s1 = 0

    try:
        s2 = int(score2)
    except Exception:
        s2 = 0

    target = parse_match_target(match_info)
    if not target:
        return None

    if s1 < target and s2 < target:
        return None

    if s1 == s2:
        return None

    if s1 >= target and s1 > s2:
        return {
            "winner_num": 1,
            "winner_name": player1 or "JOUEUR 1",
            "target": target,
            "score1": s1,
            "score2": s2
        }

    if s2 >= target and s2 > s1:
        return {
            "winner_num": 2,
            "winner_name": player2 or "JOUEUR 2",
            "target": target,
            "score1": s1,
            "score2": s2
        }

    return None

def mark_resync_needed(state, delay_seconds=0.0):
    state["force_resync"] = True
    state["next_resync_at"] = time.monotonic() + max(0.0, float(delay_seconds))

def clear_resync_needed(state):
    state["force_resync"] = False
    state["next_resync_at"] = 0.0

def clean_cuescore_name(player_obj):
    if not player_obj:
        return ""

    # Use first name only and match the website's 15-character uppercase limit.
    name = player_obj.get("firstname") or player_obj.get("name") or ""
    name = clean_single_line(name).upper()

    if PLAYER_NAME_MAX_CHARS and len(name) > PLAYER_NAME_MAX_CHARS:
        name = name[:PLAYER_NAME_MAX_CHARS].rstrip()

    return name

def latest_breaker_from_cuescore_notes(notes):
    for note in reversed(notes or []):
        text = str(note.get("note", "")).strip()
        if text == "A breaking":
            return "1"
        if text == "B breaking":
            return "2"

    return "0"

def build_display(player1, score1, score2, player2):
    return f"{player1} {score1} - {score2} {player2}".strip()

def fetch_cuescore_data(cfg):
    table_id = str(cfg.get("cuescore_table_id", "")).strip()
    if not table_id:
        return None

    return http_post_json(
        CUESCORE_ENDPOINT,
        {
            "tableId": table_id,
            "lang": "fr",
        },
        headers={
            "Referer": f"https://cuescore.com/scoreboard/overlay/?tableId={table_id}",
        },
        timeout=HTTP_TIMEOUT_CUESCORE
    )

def extract_cuescore_state(data):
    if not data:
        return None

    match = data.get("match") or {}

    status = str(data.get("status", "")).lower()
    matchstatus = str(match.get("matchstatus", "")).lower()

    if status != "playing" or matchstatus != "playing":
        return None

    player1 = clean_cuescore_name(match.get("playerA") or {})
    player2 = clean_cuescore_name(match.get("playerB") or {})

    score1 = score_to_text(match.get("scoreA", 0))
    score2 = score_to_text(match.get("scoreB", 0))

    race_to = match.get("raceTo", "")
    match_info = f"Premier à {race_to}" if str(race_to).strip() else DEFAULT_MATCH_INFO
    match_info_enabled = "1" if match_info else "0"

    breaker = latest_breaker_from_cuescore_notes(match.get("notes", []))

    return {
        "source": "cuescore",
        "visible": "1",
        "breaker": breaker,
        "auto_reset_cancel": "0",
        "player1": player1,
        "player2": player2,
        "score1": score1,
        "score2": score2,
        "display": build_display(player1, score1, score2, player2),
        "match_info_enabled": match_info_enabled,
        "match_info": match_info,
        "match_id": str(match.get("matchId", "")),
        "cuescore_version": str(match.get("curVersion", "")),
    }

def update_cuescore_cache_if_due(cfg, state, loop_started):
    if not cfg.get("cuescore_enabled", False):
        state["cuescore_active"] = False
        state["cuescore_state"] = None
        return

    poll_interval = CUESCORE_ACTIVE_POLL_SECONDS if state.get("cuescore_active") else CUESCORE_IDLE_POLL_SECONDS

    if loop_started - state["last_cuescore_poll_at"] < poll_interval:
        return

    state["last_cuescore_poll_at"] = loop_started

    try:
        data = fetch_cuescore_data(cfg)
        cuescore_state = extract_cuescore_state(data)

        was_active = state.get("cuescore_active", False)
        now_active = bool(cuescore_state)

        state["cuescore_active"] = now_active
        state["cuescore_state"] = cuescore_state

        if now_active and not was_active:
            log_table(cfg, "CueScore match detected. CueScore now controls the scoreboard.")
            clear_pending_auto_reset(state)
            mark_resync_needed(state, 0.0)

        elif (not now_active) and was_active:
            log_table(cfg, "CueScore match ended/disappeared. Falling back to website dweet control.")
            clear_pending_auto_reset(state)
            mark_resync_needed(state, 0.0)

    except Exception as e:
        log_table(cfg, f"CueScore check error: {e}")
        # Keep previous CueScore state during temporary errors.
        # This avoids website/dweet accidentally taking over mid-match.

def pause_cuescore_while_camera_offline(cfg, state):
    if state.get("cuescore_active") or state.get("cuescore_state") is not None:
        log_table(cfg, "Camera offline - pausing CueScore polling/control.")
        state["cuescore_active"] = False
        state["cuescore_state"] = None
        state["last_seen_signature"] = None
        state["last_source"] = "dweet"
        clear_pending_auto_reset(state)

    # Force an immediate CueScore check as soon as the camera comes back online.
    state["last_cuescore_poll_at"] = 0.0

def camera_request(cfg, params):
    query = build_camera_query(params)
    url = f"{cfg['camera_base']}?{query}"
    try:
        with cfg["camera_opener"].open(url, timeout=HTTP_TIMEOUT_CAMERA) as response:
            return response.read().decode("utf-8", errors="replace")
    except urllib.error.HTTPError as e:
        try:
            body = e.read().decode("utf-8", errors="replace")
        except Exception:
            body = ""
        log_table(cfg, f"Camera HTTP error {e.code}: {body}")
        return ""
    except Exception as e:
        log_table(cfg, f"Camera request error: {e}")
        return ""

def camera_is_online(cfg):
    raw = camera_request(cfg, {
        "action": "getConfig",
        "name": "ChannelTitle[0].Name"
    })

    if not raw or not raw.strip():
        return False

    low = raw.lower()
    if "unauthorized" in low:
        return False

    return True

def camera_set_config(cfg, params, label):
    query = build_camera_query(params)
    url = f"{cfg['camera_base']}?{query}"

    try:
        with cfg["camera_opener"].open(url, timeout=HTTP_TIMEOUT_CAMERA) as response:
            raw = response.read().decode("utf-8", errors="replace")
            log_table(cfg, f"{label} response: {raw.strip()}")
            return SUCCESS_TEXT in raw
    except urllib.error.HTTPError as e:
        try:
            body = e.read().decode("utf-8", errors="replace")
        except Exception:
            body = ""
        log_table(cfg, f"{label} HTTP error {e.code}: {body}")
        return False
    except Exception as e:
        log_table(cfg, f"{label} error: {e}")
        return False

def clear_pending_auto_reset(state):
    state["pending_auto_reset_deadline_ms"] = None
    state["pending_auto_reset_key"] = None
    state["pending_auto_reset_cancelled"] = False

def get_dweet_control_state(cfg, state):
    latest = fetch_latest_dweet(cfg)
    if not latest:
        log_table(cfg, "No dweet received.")
        return None

    visible = str(latest.get("visible", "1"))
    breaker = str(latest.get("breaker", "1")).strip()
    auto_reset_cancel = str(latest.get("auto_reset_cancel", "0")).strip()

    player1 = clean_single_line(get_first_nonempty(latest, ["player1", "name1"], ""))
    player2 = clean_single_line(get_first_nonempty(latest, ["player2", "name2"], ""))

    score1_raw = get_first_nonempty(latest, ["score1", "p1score", "player1score"], "")
    score2_raw = get_first_nonempty(latest, ["score2", "p2score", "player2score"], "")

    display = str(latest.get("display", ""))

    if score1_raw == "" or score2_raw == "":
        parsed1, parsed2 = parse_scores_from_display(display)
        if score1_raw == "":
            score1_raw = parsed1
        if score2_raw == "":
            score2_raw = parsed2

    score1 = score_to_text(score1_raw)
    score2 = score_to_text(score2_raw)

    match_info_enabled = str(
        get_first_nonempty(latest, ["match_info_enabled", "match_title_enabled"], "0")
    )
    match_info = clean_single_line(
        get_first_nonempty(latest, ["match_info", "match_title"], DEFAULT_MATCH_INFO)
    )

    ts = safe_int(latest.get("ts"), 0)
    if ts > 0:
        state["last_seen_ts"] = ts

    return {
        "source": "dweet",
        "visible": visible,
        "breaker": breaker,
        "auto_reset_cancel": auto_reset_cancel,
        "player1": player1,
        "player2": player2,
        "score1": score1,
        "score2": score2,
        "display": display,
        "match_info_enabled": match_info_enabled,
        "match_info": match_info,
        "match_id": "",
        "cuescore_version": "",
    }

def process_control_state(cfg, state, control):
    if not control:
        return

    source = control.get("source", "dweet")
    visible = str(control.get("visible", "1"))
    breaker = str(control.get("breaker", "1")).strip()
    auto_reset_cancel = str(control.get("auto_reset_cancel", "0")).strip()

    player1 = clean_single_line(control.get("player1", ""))
    player2 = clean_single_line(control.get("player2", ""))

    score1 = score_to_text(control.get("score1", "0"))
    score2 = score_to_text(control.get("score2", "0"))

    match_info_enabled = str(control.get("match_info_enabled", "0"))
    match_info = clean_single_line(control.get("match_info", DEFAULT_MATCH_INFO))

    player1_plain = build_player_line(score1, player1, False)
    player2_plain = build_player_line(score2, player2, False)
    player1_break = build_player_line(score1, player1, breaker == "1")
    player2_break = build_player_line(score2, player2, breaker == "2")

    if breaker == "1":
        player1_final = player1_break
        player2_final = player2_plain
    elif breaker == "2":
        player1_final = player1_plain
        player2_final = player2_break
    else:
        player1_final = player1_plain
        player2_final = player2_plain

    name_box_rect1, name_box_rect2 = get_name_box_rects(player1, player2)

    signature = build_state_signature(
        visible, breaker, auto_reset_cancel,
        player1, player2,
        score1, score2,
        match_info_enabled, match_info
    )

    # Switching between CueScore and website/dweet should force a clean update.
    if source != state.get("last_source"):
        log_table(cfg, f"Control source switched to {source}.")
        state["last_seen_signature"] = None
        state["last_source"] = source

    if signature != state["last_activity_signature"]:
        state["last_activity_signature"] = signature
        state["last_activity_monotonic"] = time.monotonic()

    winner_info = None
    winner_key = None

    # Do not auto-reset CueScore-controlled matches. CueScore decides when a match finishes/changes.
    if source != "cuescore" and visible == "1" and match_info_enabled == "1":
        winner_info = get_winner_info(score1, score2, match_info, player1, player2)

    if winner_info:
        winner_key = (
            f"{winner_info['winner_num']}|{winner_info['target']}|"
            f"{winner_info['score1']}|{winner_info['score2']}|"
            f"{player1}|{player2}|{match_info}"
        )

        if auto_reset_cancel == "1":
            if state["pending_auto_reset_key"] != winner_key:
                state["pending_auto_reset_key"] = winner_key
                state["pending_auto_reset_deadline_ms"] = None
            state["pending_auto_reset_cancelled"] = True
            log_table(cfg, "Auto reset cancelled by website.")
        else:
            if state["pending_auto_reset_key"] != winner_key or state["pending_auto_reset_deadline_ms"] is None:
                state["pending_auto_reset_key"] = winner_key
                state["pending_auto_reset_deadline_ms"] = now_ms() + (AUTO_RESET_SECONDS * 1000)
                state["pending_auto_reset_cancelled"] = False
                log_table(
                    cfg,
                    f"Winner detected: {winner_info['winner_name']} "
                    f"(target {winner_info['target']}) - auto reset in {AUTO_RESET_SECONDS}s."
                )
    else:
        clear_pending_auto_reset(state)

    force_resync_due = (
        state.get("force_resync", False) and
        time.monotonic() >= state.get("next_resync_at", 0.0)
    )

    if force_resync_due and state.get("last_camera_online") is False:
        return

    if signature == state["last_seen_signature"] and not force_resync_due:
        return

    was_visible = (state["last_visible_value"] == "1")
    now_visible = (visible == "1")

    state["last_seen_signature"] = signature
    state["last_visible_value"] = visible

    if force_resync_due:
        log_table(cfg, "Forcing scoreboard resync to camera.")

    if state.get("last_camera_online") is False:
        mark_resync_needed(state, 0.0)
        log_table(cfg, "Camera offline - state recorded; resync will run when camera returns.")
        return

    if now_visible and not was_visible:
        ok_prep1 = prepare_channel_title_line(cfg, player1_final)
        ok_prep2 = prepare_player2_line(cfg, player2_final)

        if match_info_enabled == "1" and match_info.strip():
            ok_prep_info = prepare_match_info(cfg, match_info)
        else:
            ok_prep_info = hide_match_info(cfg)

        p1_has_name = bool(player1)
        p2_has_name = bool(player2)

        ok_red = set_cover_box(cfg, PLAYER1_BOX_INDEX, PLAYER1_BOX_RECT, PLAYER1_BOX_COLOR, 1)

        if p1_has_name:
            ok_black1 = show_name_box(cfg, PLAYER1_NAME_BOX_INDEX, name_box_rect1)
        else:
            ok_black1 = hide_name_box(cfg, PLAYER1_NAME_BOX_INDEX, name_box_rect1)

        ok_p1 = show_channel_title_line(cfg)

        ok_blue = set_cover_box(cfg, PLAYER2_BOX_INDEX, PLAYER2_BOX_RECT, PLAYER2_BOX_COLOR, 1)

        if p2_has_name:
            ok_black2 = show_name_box(cfg, PLAYER2_NAME_BOX_INDEX, name_box_rect2)
        else:
            ok_black2 = hide_name_box(cfg, PLAYER2_NAME_BOX_INDEX, name_box_rect2)

        ok_p2 = show_player2_line(cfg)

        if match_info_enabled == "1" and match_info.strip():
            ok_info = show_match_info(cfg)
        else:
            ok_info = True

        success = (
            ok_prep1 and ok_prep2 and ok_prep_info and ok_red and ok_black1 and
            ok_p1 and ok_blue and ok_black2 and ok_p2 and ok_info
        )

        if success:
            clear_resync_needed(state)
            log_table(cfg, f"Scoreboard shown from {source}: {player1_final} / {player2_final} / {match_info}")
        else:
            mark_resync_needed(state, RESYNC_RETRY_SECONDS)
            log_table(cfg, "Failed to show one or more scoreboard elements.")

    elif (not now_visible) and was_visible:
        clear_pending_auto_reset(state)
        success = clear_overlay(cfg, player1, player2)

        if success:
            clear_resync_needed(state)
            log_table(cfg, "Scoreboard hidden.")
        else:
            mark_resync_needed(state, RESYNC_RETRY_SECONDS)
            log_table(cfg, "Failed to hide one or more scoreboard elements.")

    elif now_visible:
        p1_has_name = bool(player1)
        p2_has_name = bool(player2)

        if p1_has_name:
            ok_black1 = show_name_box(cfg, PLAYER1_NAME_BOX_INDEX, name_box_rect1)
        else:
            ok_black1 = hide_name_box(cfg, PLAYER1_NAME_BOX_INDEX, name_box_rect1)

        if p2_has_name:
            ok_black2 = show_name_box(cfg, PLAYER2_NAME_BOX_INDEX, name_box_rect2)
        else:
            ok_black2 = hide_name_box(cfg, PLAYER2_NAME_BOX_INDEX, name_box_rect2)

        if breaker == "1":
            ok_p2 = set_player2_line(cfg, player2_plain, 1)
            ok_p1 = set_channel_title_line(cfg, player1_break, 1)
        elif breaker == "2":
            ok_p1 = set_channel_title_line(cfg, player1_plain, 1)
            ok_p2 = set_player2_line(cfg, player2_break, 1)
        else:
            ok_p1 = set_channel_title_line(cfg, player1_plain, 1)
            ok_p2 = set_player2_line(cfg, player2_plain, 1)

        ok_red = set_cover_box(cfg, PLAYER1_BOX_INDEX, PLAYER1_BOX_RECT, PLAYER1_BOX_COLOR, 1)
        ok_blue = set_cover_box(cfg, PLAYER2_BOX_INDEX, PLAYER2_BOX_RECT, PLAYER2_BOX_COLOR, 1)

        if match_info_enabled == "1" and match_info.strip():
            ok_info = set_match_info(cfg, match_info, 1)
        else:
            ok_info = hide_match_info(cfg)

        success = ok_black1 and ok_black2 and ok_p1 and ok_p2 and ok_red and ok_blue and ok_info

        if success:
            clear_resync_needed(state)
            log_table(cfg, f"Scoreboard updated from {source}: {player1_final} / {player2_final} / {match_info}")
        else:
            mark_resync_needed(state, RESYNC_RETRY_SECONDS)
            log_table(cfg, "Failed to update one or more scoreboard elements.")

    else:
        clear_pending_auto_reset(state)
        success = clear_overlay(cfg, player1, player2)

        if success:
            clear_resync_needed(state)
            log_table(cfg, "Scoreboard remains hidden.")
        else:
            mark_resync_needed(state, RESYNC_RETRY_SECONDS)
            log_table(cfg, "Failed to maintain hidden scoreboard state.")

def auto_reset_check(cfg, state):
    if state.get("cuescore_active"):
        return

    if state["pending_auto_reset_deadline_ms"] is None:
        return

    if state["pending_auto_reset_cancelled"]:
        return

    if now_ms() < state["pending_auto_reset_deadline_ms"]:
        return

    log_table(cfg, "Winner countdown elapsed. Auto-clearing overlay.")
    clear_ok = clear_overlay(cfg, "", "")

    latest = fetch_latest_dweet(cfg) or {}

    reset_player1 = clean_single_line(get_first_nonempty(latest, ["player1", "name1"], ""))
    reset_player2 = clean_single_line(get_first_nonempty(latest, ["player2", "name2"], ""))
    reset_match_title_enabled = str(
        get_first_nonempty(latest, ["match_info_enabled", "match_title_enabled"], "0")
    )
    reset_match_title = clean_single_line(
        get_first_nonempty(latest, ["match_info", "match_title"], DEFAULT_MATCH_INFO)
    )

    dweet_reset_state(
        cfg,
        player1=reset_player1,
        player2=reset_player2,
        match_title_enabled=reset_match_title_enabled,
        match_title=reset_match_title
    )

    state["last_seen_signature"] = None
    state["last_visible_value"] = "0"
    state["last_seen_ts"] = now_ms()
    state["last_activity_signature"] = build_state_signature(
        "0", "1", "0",
        reset_player1, reset_player2,
        "0", "0",
        reset_match_title_enabled,
        reset_match_title
    )
    state["last_activity_monotonic"] = time.monotonic()

    if clear_ok:
        clear_resync_needed(state)
    else:
        mark_resync_needed(state, RESYNC_RETRY_SECONDS)

    clear_pending_auto_reset(state)

def idle_check(cfg, state):
    if state.get("cuescore_active"):
        return

    if state["last_activity_monotonic"] <= 0:
        return

    age_seconds = time.monotonic() - state["last_activity_monotonic"]
    if age_seconds < IDLE_RESET_SECONDS:
        return

    log_table(cfg, "Inactivity detected. Auto-clearing overlay.")
    clear_ok = clear_overlay(cfg, "", "")

    latest = fetch_latest_dweet(cfg) or {}

    reset_player1 = clean_single_line(get_first_nonempty(latest, ["player1", "name1"], ""))
    reset_player2 = clean_single_line(get_first_nonempty(latest, ["player2", "name2"], ""))
    reset_match_title_enabled = str(
        get_first_nonempty(latest, ["match_info_enabled", "match_title_enabled"], "0")
    )
    reset_match_title = clean_single_line(
        get_first_nonempty(latest, ["match_info", "match_title"], DEFAULT_MATCH_INFO)
    )

    dweet_reset_state(
        cfg,
        player1=reset_player1,
        player2=reset_player2,
        match_title_enabled=reset_match_title_enabled,
        match_title=reset_match_title
    )

    state["last_seen_signature"] = None
    state["last_visible_value"] = "0"
    state["last_seen_ts"] = now_ms()
    state["last_activity_signature"] = build_state_signature(
        "0", "1", "0",
        reset_player1, reset_player2,
        "0", "0",
        reset_match_title_enabled,
        reset_match_title
    )
    state["last_activity_monotonic"] = time.monotonic()

    if clear_ok:
        clear_resync_needed(state)
    else:
        mark_resync_needed(state, RESYNC_RETRY_SECONDS)

    clear_pending_auto_reset(state)

def main():
    table_cfgs = [build_table_cfg(t) for t in TABLES]
    table_states = {cfg["table_key"]: build_initial_state() for cfg in table_cfgs}

    log("Starting CueScore-aware 2-table headless scoreboard controller.")

    for cfg in table_cfgs:
        try:
            if clear_overlay(cfg, "", ""):
                log_table(cfg, "Startup overlay clear complete.")
            else:
                log_table(cfg, "Startup overlay clear had one or more failures.")
        except Exception as e:
            log_table(cfg, f"Startup overlay clear error: {e}")

        table_states[cfg["table_key"]]["last_visible_value"] = "0"

    while True:
        loop_started = time.time()

        for cfg in table_cfgs:
            state = table_states[cfg["table_key"]]

            try:
                if loop_started - state["last_camera_status_push_at"] >= CAMERA_STATUS_SECONDS:
                    online = camera_is_online(cfg)

                    previous_online = state.get("last_camera_online")
                    if online and previous_online is False:
                        log_table(cfg, "Camera came back online. Forcing scoreboard and CueScore resync.")
                        state["last_cuescore_poll_at"] = 0.0
                        mark_resync_needed(state, 0.0)

                    if not online:
                        pause_cuescore_while_camera_offline(cfg, state)

                    state["last_camera_online"] = online
                    push_camera_status(cfg, state, online)
                    state["last_camera_status_push_at"] = loop_started
            except Exception as e:
                log_table(cfg, f"Camera status section error: {e}")

            try:
                if state.get("last_camera_online") is True:
                    update_cuescore_cache_if_due(cfg, state, loop_started)
                else:
                    pause_cuescore_while_camera_offline(cfg, state)
            except Exception as e:
                log_table(cfg, f"CueScore section error: {e}")

            try:
                if loop_started - state["last_dweet_poll_at"] >= POLL_SECONDS:
                    if state.get("cuescore_active") and state.get("cuescore_state"):
                        process_control_state(cfg, state, state["cuescore_state"])
                    else:
                        dweet_state = get_dweet_control_state(cfg, state)
                        process_control_state(cfg, state, dweet_state)

                    state["last_dweet_poll_at"] = loop_started
            except Exception as e:
                log_table(cfg, f"Scoreboard processing section error: {e}")

            try:
                auto_reset_check(cfg, state)
            except Exception as e:
                log_table(cfg, f"Auto reset section error: {e}")

            try:
                idle_check(cfg, state)
            except Exception as e:
                log_table(cfg, f"Idle check section error: {e}")

        time.sleep(0.2)

if __name__ == "__main__":
    main()
